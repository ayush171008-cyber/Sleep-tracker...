                db.createObjectStore("sleep_records", { keyPath: "id", autoIncrement: true });
            }
        };
        request.onsuccess = (e) => {
            db = e.target.result;
            hydrateApplication();
        };


        function switchTab(tabId) {
            document.querySelectorAll('.tab-content').forEach(el => el.classList.add('hidden'));
            document.getElementById(`tab-${tabId}`).classList.remove('hidden');
            document.querySelectorAll('.nav-btn').forEach(btn => btn.classList.remove('text-indigo-400'));
            event.currentTarget.classList.add('text-indigo-400');

        function hydrateApplication() {
            const transaction = db.transaction(["sleep_records"], "readonly");
            const store = transaction.objectStore("sleep_records");
            const getAll = store.getAll();
            getAll.onsuccess = () => {
                cachedRecords = getAll.result.sort((a,b) => new Date(b.date + 'T' + b.bedtime) - new Date(a.date + 'T' + a.bedtime));
                // Safety Backup Routine
                localStorage.setItem('somnus_backup', JSON.stringify(cachedRecords));
                calculateCoreMetrics();
                renderHistory();
                buildDashboardChart();
            };
        }


        function calculateCoreMetrics() {
            if(cachedRecords.length === 0) return;
            
            const last = cachedRecords[0];
            const lastDuration = calculateDuration(last.bedtime, last.wakeup);
            document.getElementById('stat-last-duration').innerText = lastDuration.toFixed(1);
            
            const score = Math.min(100, Math.round((lastDuration / targets.duration) * 50 + (last.quality * 5)));
            const badge = document.getElementById('stat-last-score-badge');
            badge.innerText = `Sleep Score: ${score}/100`;
            
            // Average Calculations
            let totalDur = 0; let totalQual = 0;
            cachedRecords.forEach(r => {
                totalDur += calculateDuration(r.bedtime, r.wakeup);
                totalQual += parseInt(r.quality);
            });
            const avgDur = totalDur / cachedRecords.length;
            const avgQual = totalQual / cachedRecords.length;
            
            document.getElementById('stat-avg-quality').innerText = avgQual.toFixed(1);
            const debt = (targets.duration * cachedRecords.length) - totalDur;
            document.getElementById('stat-sleep-debt').innerText = Math.max(0, debt).toFixed(1);


            // Ring progress calculation
            const pct = Math.min(100, Math.round((avgDur / targets.duration) * 100));
            document.getElementById('goal-pct-text').innerText = `${pct}%`;
            const offset = 251.2 - (251.2 * pct) / 100;
            document.getElementById('goal-progress-ring').style.strokeDashoffset = offset;


            // Generate Insights 
            const insightsBox = document.getElementById('insights-container');
            if (cachedRecords.length >= 2) {
                insightsBox.innerHTML = `
                    <div class="flex items-center space-x-2 text-emerald-400 font-medium">
                        <span class="material-icons-round text-sm">trending_up</span>
                        <span>Your baseline analytical aggregate is hovering at ${avgDur.toFixed(1)} hrs per evening structure.</span>
                    </div>
                    <div class="text-slate-400 pl-6 text-xs">Target validation indicates an optimization alignment of ${pct}% against parameters.</div>
                `;
            }
            
            // Simple operational streaks engine
            document.getElementById('streak-display').innerText = `${cachedRecords.length} Day Logged`;
        }


        function calculateDuration(bedtime, wakeup) {
            let [bH, bM] = bedtime.split(':').map(Number);
            let [wH, wM] = wakeup.split(':').map(Number);
            let bDate = new Date(2020, 0, 1, bH, bM);
            let wDate = new Date(2020, 0, 1, wH, wM);
            if (wDate < bDate) wDate.setDate(wDate.getDate() + 1);
            return (wDate - bDate) / 1000 / 60 / 60;
        }

        function saveEntry(e) {
            e.preventDefault();
            const id = document.getElementById('entry-id').value;
            const record = {
                date: document.getElementById('form-date').value,
                quality: document.getElementById('form-quality').value,
                bedtime: document.getElementById('form-bedtime').value,
                wakeup: document.getElementById('form-wakeup').value,
                mood: document.getElementById('form-mood').value,
                notes: document.getElementById('form-notes').value
            };
            if(id) record.id = Number(id);


            const transaction = db.transaction(["sleep_records"], "readwrite");
            const store = transaction.objectStore("sleep_records");
            const request = id ? store.put(record) : store.add(record);


            request.onsuccess = () => {
                closeModal();
                hydrateApplication();
            };
        }


        function deleteEntry(id) {
            if(!confirm("Are you sure you want to delete this specific sleep record?")) return;

            const store = transaction.objectStore("sleep_records");
            store.delete(id).onsuccess = () => hydrateApplication();
        }

        function importJSON(e) {
            const reader = new FileReader();
            reader.onload = (event) => {
                const data = JSON.parse(event.target.result);
                const tx = db.transaction(["sleep_records"], "readwrite");
                const store = tx.objectStore("sleep_records");
                data.forEach(item => { delete item.id; store.add(item); });
                tx.oncomplete = () => { hydrateApplication(); alert("Import matrix integrated successfully!"); };
            };
            reader.readAsText(e.target.files[0]);
        }
